
            {* Project Euler problem number 1 *}
            { Find the sum of the multiples of 3 or 5 below 1000 }
            function proj_euler_1(): integer;
            var sum, i: integer;
            begin
                sum := 0;
                for i := 0 to 999 do
                    if (i mod 3 = 0) or (i mod 5 = 0) then sum := sum + i;
            end
